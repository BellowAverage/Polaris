#### 雪花

雪落效果主要代码实现如下：

```
screen.delay(0)
t = Turtle(visible = False,shape='circle')
t.pencolor("white")
t.fillcolor("white")
t.penup()
t.setheading(-90)
t.goto(r.randint(-width/2,width/2),height/2)
stars = []
for i in range(200):
    star = t.clone()
    s =r.random() / 3
    star.shapesize(s,s)
    star.speed(int(s*10))
    star.setx(r.randint(-width/2,width/2))
    star.sety(height/2 + r.randint(1,height))
    star.showturtle()
    stars.append(star)
while True:
    for star in stars:
        star.sety(star.ycor() - 8 * star.speed())
        if star.ycor()&lt;-height/2:
            star.hideturtle()
            star.setx(r.randint(-width/2,width/2))
            star.sety(height/2 + r.randint(1,height))
            star.showturtle()

```
